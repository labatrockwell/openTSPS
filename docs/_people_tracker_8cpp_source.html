00708     //---------------------------------------------------------------------------
00709     void PeopleTracker::trackPeople()
00710     {   
00711         if (p_Settings == NULL) p_Settings = gui.getSettings();
00712         
00713         //-------------------------------
00714         // Camera view
00715         //-------------------------------
00716         
00717         //warp background
00718         //colorImage = grayImage;
00719         //colorImageWarped = colorImage;
00720         getQuadSubImage(cameraImage, warpedImage, p_Settings->quadWarpScaled, OF_IMAGE_GRAYSCALE);      
00721         
00722         // mirror?
00723         if ( p_Settings->bFlipX || p_Settings->bFlipY ){
00724             int mode = 1;
00725             if ( p_Settings->bFlipX && p_Settings->bFlipY ) mode = -1;
00726             else if ( p_Settings->bFlipY ) mode = 0;
00727             
00728             cv::Mat src = ofxCv::toCv( warpedImage ), dst = ofxCv::toCv( warpedImage );
00729             cv::flip( src, dst, mode );
00730             ofxCv::toOf( dst, warpedImage );
00731             warpedImage.update();
00732         }
00733         
00734         // invert
00735         if ( p_Settings->bInvert ){
00736             ofPixelsRef pix = warpedImage.getPixelsRef();
00737             for (int i=0, len=pix.getWidth()*pix.getHeight()*pix.getNumChannels();i<len;i++){
00738                 pix[i]=255-pix[i];
00739             }
00740             warpedImage.update();
00741         }
00742         
00743         // update scaled down images
00744         grayDiff.setFromPixels(warpedImage.getPixelsRef());
00745         //grayDiff = grayImageWarped;
00746         
00747         //amplify
00748         if(p_Settings->bAmplify){
00749             grayDiff.amplify(grayDiff, p_Settings->highpassAmp/15.0f);
00750             warpedImage.setFromPixels(grayDiff.getPixelsRef());
00751         }
00752         
00753         // set base camera image
00754         tspsProcessor->setCameraImage( warpedImage );
00755         
00756         //learn background
00757         if (doRelearnBackground){
00758             backgroundImage.setFromPixels(warpedImage.getPixelsRef());
00759             tspsProcessor->captureBackground( warpedImage );
00760             doRelearnBackground = false;
00761         }
00762         
00763         //progressive relearn background
00764         if (p_Settings->bLearnBackgroundProgressive){
00765             backgroundImage.setFromPixels( tspsProcessor->progressiveBackground( warpedImage, p_Settings->fLearnRate * .00001 ));
00766         }
00767         
00768         // black out background?
00769         if (p_Settings->bBlankBackground){
00770             tspsProcessor->blankBackground();
00771             backgroundImage.setFromPixels(blackPixels);
00772         }
00773         
00774         //-----------------------
00775         // Difference image
00776         //-----------------------
00777         grayDiff.setFromPixels( tspsProcessor->difference( grayDiff, (TrackingType) p_Settings->trackType ) );
00778         
00779         //-----------------------
00780         // Post-difference filters
00781         //-----------------------
00782         
00783         if(p_Settings->bSmooth){
00784             grayDiff.blur((p_Settings->smooth * 2) + 1); //needs to be an odd number
00785         }
00786         
00787         //highpass filter
00788         if(p_Settings->bHighpass){
00789             grayDiff.highpass(p_Settings->highpassBlur, p_Settings->highpassNoise);
00790         }
00791         
00792         //-----------------------
00793         // Track
00794         //-----------------------       
00795         differencedImage.setFromPixels( tspsProcessor->process( grayDiff ) );
00796         
00797         //-----------------------
00798         // Custom properties
00799         //-----------------------
00800         
00801         // add depth if we've got it
00802          for(int i = 0; i < trackedPeople.size(); i++){
00803              ofxTSPS::Person* p = trackedPeople[i];
00804              if (currentSource->getType() == CAMERA_KINECT){
00805                  // distance is in mm, with the max val being 10 m
00806                  // scale it by max to get it in a 0-1 range
00807                  p->depth = ((ofxKinect*)currentSource)->getDistanceAt( p->highest )/10000.0;
00808              } else {
00809                  p->depth = p->highest.z / 255.0f;
00810              }
00811          }
00812     }
